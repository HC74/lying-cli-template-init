项目结构:

```
├── api  
│ ├── api.js                    // 接口调用方法  
│ ├── myFetch.js                // 封装的请求实例  
│ └── totalUrl.js               // 接口地址
├── components                  // 公共组件
├── constant                    // 常量
│ ├── config.js                 // 配置(开发、生产环境等地址)
├── stores                      // 状态管理
│ ├── index.js                  // pinia
├── utils                       // 工具集
│ ├── index.js                  // pinia
│ ├── auth.js                   // token管理  
│ ├── router-intercept.js       // 路由拦截
├── uni_modules                 // 插件市场插件目录
│ ├── uview-plus-ui             // uview-plus-ui
│ ├── uni-ui(uni-xx)            // uni-ui    
│ ├── z-paging                  // 下拉刷新、上拉加载更多
```
